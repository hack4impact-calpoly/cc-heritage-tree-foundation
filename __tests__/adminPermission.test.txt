/**
 * @jest-environment jsdom
 */

import React from "react";
import "@testing-library/jest-dom";
import { render, screen } from "@testing-library/react";
import Messages from "@/app/messages/page";
import Navbar from "@/components/Navbar";

jest.mock("@clerk/nextjs", () => ({
  ...jest.requireActual("@clerk/nextjs"),
  useUser: () => ({
    user: {
      id: "user_123",
      fullName: "Test User",
      primaryEmailAddress: {
        emailAddress: "janedoe123@gmail.com",
      },
    },
    isLoaded: true,
  }),
}));

jest.mock("@clerk/nextjs", () => {
  const actual = jest.requireActual("@clerk/nextjs");
  return {
    ...actual,
    ClerkProvider: ({ children }: { children: React.ReactNode }) => <div>{children}</div>,
    useUser: jest.fn(),
  };
});

jest.mock("next/navigation", () => ({
  useRouter: () => ({
    push: jest.fn(),
  }),
}));

const mockUseUser = require("@clerk/nextjs").useUser;

afterEach(() => {
  jest.resetAllMocks();
});

describe("Messages Page", () => {
  it("shows 'New Message +' button for admin users", () => {
    mockUseUser.mockReturnValue({
      isLoaded: true,
      user: {
        organizationMemberships: [{ role: "org:admin" }],
      },
    });

    render(<Messages />);
    expect(screen.getByText("New Message +")).toBeInTheDocument();
  });

  it("does NOT show 'New Message +' button for volunteers", () => {
    mockUseUser.mockReturnValue({
      isLoaded: true,
      user: {
        organizationMemberships: [{ role: "org:volunteer" }],
      },
    });

    render(<Messages />);
    expect(screen.queryByText("New Message +")).not.toBeInTheDocument();
  });
});

describe("Navbar Role Filtering", () => {
  it("shows 'Volunteers' path for admin users", () => {
    mockUseUser.mockReturnValue({
      isLoaded: true,
      user: {
        organizationMemberships: [{ role: "org:admin" }],
      },
    });

    render(<Navbar />);
    expect(screen.getByText("Volunteers")).toBeInTheDocument();
  });

  it("hides 'Volunteers' path for non-admin users", () => {
    mockUseUser.mockReturnValue({
      isLoaded: true,
      user: {
        organizationMemberships: [{ role: "org:volunteer" }],
      },
    });

    render(<Navbar />);
    expect(screen.queryByText("Volunteers")).not.toBeInTheDocument();
  });
});
