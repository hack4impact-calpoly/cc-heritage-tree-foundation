/**
 * @jest-environment jsdom
 */

import { render, screen, fireEvent, waitFor } from "@testing-library/react";
import Navbar from "@/components/Navbar";
import { ChakraProvider } from "@chakra-ui/react";
import { isMobile } from "react-device-detect";
import "@testing-library/jest-dom";
import { ClerkProvider } from "@clerk/nextjs";
import React, { ReactNode } from "react";

// Improved mock for next/navigation
jest.mock('next/navigation', () => ({
  useRouter: () => ({
    push: jest.fn(),
    replace: jest.fn(),
    refresh: jest.fn(),
    prefetch: jest.fn(),
    back: jest.fn(),
    forward: jest.fn(),
  }),
  usePathname: () => '/',
  useSearchParams: () => new URLSearchParams(),
}));

// Mock react-device-detect
jest.mock("react-device-detect", () => ({
  isMobile: true,
}));

// Mock Clerk
jest.mock("@clerk/nextjs", () => ({
  ClerkProvider: ({ children }: { children: ReactNode }) => <>{children}</>,
  useUser: () => ({
    user: {
      id: "test-user-id",
      primaryEmailAddress: "test@example.com",
      fullName: "Test User",
      organizationMemberships: [{ role: "org:admin" }],
    },
    isLoaded: true,
    isSignedIn: true,
  }),
}));

describe("Navbar component", () => {
  const pushMock = jest.fn();

  beforeEach(() => {
    // Clear all mocks between tests
    jest.clearAllMocks();
    
    // Mock the router push function
    require('next/navigation').useRouter.mockImplementation(() => ({
      push: pushMock,
    }));
  });

  test("renders the mobile navbar and opens the drawer", async () => {
    render(
      <ChakraProvider>
        <Navbar />
      </ChakraProvider>
    );

    const menuButton = screen.getByLabelText("Open Navigation");
    fireEvent.click(menuButton);

    await waitFor(() => {
      expect(screen.getByText("Dashboard")).toBeInTheDocument();
      expect(screen.getByText("Tree Inventory")).toBeInTheDocument();
      expect(screen.getByText("Messages")).toBeInTheDocument();
      expect(screen.getByText("Map")).toBeInTheDocument();
    });
  });

  test("navigates correctly when a nav item is clicked", async () => {
    render(
      <ChakraProvider>
        <ClerkProvider>
          <Navbar />
        </ClerkProvider>
      </ChakraProvider>
    );

    fireEvent.click(screen.getByLabelText(/Open Navigation/i));
    fireEvent.click(screen.getByText("Tree Inventory"));

    expect(pushMock).toHaveBeenCalledWith("/treeTable");
  });

  test("doesn't show Volunteers tab for non-admin users", async () => {
    // Override the mock for non-admin user
    jest.spyOn(require('@clerk/nextjs'), 'useUser').mockReturnValueOnce({
      user: {
        organizationMemberships: [{ role: "org:member" }],
      },
      isLoaded: true,
    });

    render(
      <ChakraProvider>
        <Navbar />
      </ChakraProvider>
    );

    fireEvent.click(screen.getByLabelText(/Open Navigation/i));
    
    await waitFor(() => {
      expect(screen.queryByText("Volunteers")).not.toBeInTheDocument();
    });
  });
});
